%\def\statement#1#2{\slightnegindent \claimitem{#1}{#2}}

\long\def\statement{%
    \futurelet\statementToken\statementDecide
}

\long\def\statementDecide{%
    \ifx\statementToken [%
        \let\next = \statementstandard
    \else
        \let\next = \statementshort
    \fi
    \next
}
